function showDashboard(user) {
    const app = document.getElementById('app');
    const userRole = user.user_metadata.role;

    app.innerHTML = `
        <div class="dashboard-container">
            <header class="dashboard-header">
                <h1>نظام الرصد الإعلامي</h1>
                <button id="logout-btn">تسجيل الخروج</button>
            </header>
            <nav class="dashboard-nav">
                <a href="#" id="monitoring-link">الرصد</a>
                <a href="#" id="assignments-link">التكليفات</a>
                <a href="#" id="media-management-link">إدارة الوسائل</a>
            </nav>
            <main id="dashboard-content"></main>
        </div>
    `;

    document.getElementById('logout-btn').addEventListener('click', handleLogout);
    document.getElementById('monitoring-link').addEventListener('click', (e) => { e.preventDefault(); showMonitoringPage(userRole); });
    document.getElementById('assignments-link').addEventListener('click', (e) => { e.preventDefault(); document.getElementById('dashboard-content').innerHTML = '<h2>صفحة التكليفات (قيد التطوير)</h2>'; });
    document.getElementById('media-management-link').addEventListener('click', (e) => { e.preventDefault(); document.getElementById('dashboard-content').innerHTML = '<h2>صفحة إدارة الوسائل (قيد التطوير)</h2>'; });

    showMonitoringPage(userRole);
}

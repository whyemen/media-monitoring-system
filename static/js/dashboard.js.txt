function initializeDashboard(userData) {
    const dashboardContent = document.getElementById('dashboard-content');
    const welcomeMessage = document.getElementById('welcome-message');

    // Clear previous content
    dashboardContent.innerHTML = '';

    // Set welcome message
    welcomeMessage.textContent = `أهلاً بك، ${userData.name}`;

    // Load content based on user role
    if (userData.role === 'monitor') {
        loadMonitorDashboard(dashboardContent, userData);
    } else if (userData.role === 'analyst') {
        loadAnalystDashboard(dashboardContent);
    } else if (userData.role === 'admin') {
        dashboardContent.innerHTML = '<h3>لوحة تحكم مدير النظام</h3><p>سيتم إضافة وظائف مدير النظام هنا.</p>';
    }
}
